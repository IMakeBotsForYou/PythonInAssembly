# DEFINE a as the number you want to check for prime-ness
# and find the factors of.
DB input 60
PRINT [input]
EL [input] 1
ADD IP 31
MOV AX [input]
DIV 2
DB max_check [DX]
DB divider 2
MOV CX [max_check]
label:
    EL [input] [divider]
    ADD IP 21
    DIV [divider]
    EQ [EX] 0
    ADD IP 3
    ADD divider 1
    LOOP label
    ADD IP 15
PRINT [divider]
MOV AX [divider]
MOV CX [max_check]
SUB CX [AX]
check_factors:
    MOV AX [DX]
    DIV [divider]
    EQ [EX] 0
    PRINT [divider]
    EQ [EX] 0
    JUMP check_factors
    ADD divider 1
    MOV DX [AX]
    LOOP check_factors
ADD IP 1
PRINT PRIME
ADD IP 1
PRINT INVALID
